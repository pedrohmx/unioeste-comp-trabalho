STMTS -> STMTS STMT
STMTS -> ''
STMT -> DECL_STMT ;
STMT -> ATTRIB_STMT ;
STMT -> COMMAND ;
STMT -> FLOW
DECL_STMT -> type id DECL_END
DECL_END -> attrib EXPR
DECL_END -> ''
ATTRIB_STMT -> id ATTRIB_END 
ATTRIB_END -> attrib EXPR 
ATTRIB_END -> attrib_op EXPR
EXPR -> BOOL_EXPR
BOOL_EXPR -> not EXPR
BOOL_EXPR -> REL_EXPR BOOL_EXPR_C
BOOL_EXPR_C -> bool_op REL_EXPR
BOOL_EXPR_C -> ''
REL_EXPR -> ARITH_EXPR REL_EXPR_C
REL_EXPR_C -> rel_op ARITH_EXPR
REL_EXPR_C -> ''
ARITH_EXPR -> TERM ARITH_EXPR_C
ARITH_EXPR_C -> arith_op_sum TERM
ARITH_EXPR_C -> ''
TERM -> FACTOR TERM_C
TERM_C -> arith_op_mul FACTOR
TERM_C -> ''
FACTOR -> ( EXPR )
FACTOR -> VALUE
FLOW -> IF_STMT
FLOW -> WHILE_STMT
FLOW -> FOR_STMT
IF_STMT -> if ( VALUE ) { STMTS } ELSE_STMT 
ELSE_STMT -> else { STMTS } 
ELSE_STMT -> ''
WHILE_STMT -> while ( VALUE ) { STMTS } 
FOR_STMT -> for ( DECL_STMT ; VALUE ; ATTRIB_STMT ) { STMTS }
COMMAND -> read id 
COMMAND -> write VALUE WCHAIN
WCHAIN -> , VALUE WCHAIN
WCHAIN -> ''
VALUE -> id
VALUE -> LITERAL
LITERAL -> literal_int 
LITERAL -> literal_float
LITERAL -> literal_bool
LITERAL -> literal_str
